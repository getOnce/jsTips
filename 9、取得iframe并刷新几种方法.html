<body>

<div id="i1">

<iframe width="200" height="100" src="i1.html">

</iframe>

</div>

<div id="i2">

<iframe width="200" height="100" name="i2" src="i2.html">

</iframe>

</div>

<input type="button" value="refresh iframe1" onclick="refresh1()"/>

<input type="button" value="refresh iframe2" onclick="refresh2()"/>

<input type="button" value="refresh iframe1" onclick="refresh3()"/>

<input type="button" value="refresh iframe2" onclick="refresh4()"/>

<script>

function refresh1(){//通过 标签取到相应的iframe

var i = document.getElementsByTagName('iframe')[0];

i.contentWindow.location.reload();

}

function refresh2(){//通过 name取到相应的iframe

var i = document.getElementsByName('i2')[0];

i.contentWindow.location.reload();

}

function refresh3(){//通过父级元素找到相应的iframe

var i = document.getElementById('i1').getElementsByTagName('iframe')[0]

i.contentWindow.location.reload();

}

function refresh4(){//用jquery的方法取到相应的DOM对象

var i = $("#i1").get(0).getElementsByTagName('iframe')[0];

i.contentWindow.location.reload();

var i = $("div").get(0).getElementsByTagName('iframe')[0];

i.contentWindow.location.reload();

var i = $("div").get(1).getElementsByTagName('iframe')[0];

i.contentWindow.location.reload();

}

//----------------------------------------最后一个涉及到jquery对象转化为DOM对象 .get方法

</script>

</body>
